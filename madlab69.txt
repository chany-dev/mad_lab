MainActivity.java

package com.example.wallpaper_cha;
import android.app.WallpaperManager;
import android.os.Bundle;
import android.os.Handler;
import android.util.Log;
import android.view.View;
import android.widget.Button;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;
import java.io.IOException;
import java.util.Random;

public class MainActivity extends AppCompatActivity {

    private static final String TAG = "WallpaperApp";
    private Handler handler = new Handler();
    private int[] images = {R.drawable.img1, R.drawable.img2, R.drawable.img3};
    private Random random = new Random();
    private Runnable runnable;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btnStart = findViewById(R.id.btnStartWallpaper);

        btnStart.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                Log.d(TAG, "Button Clicked!");
                Toast.makeText(MainActivity.this, "Service Started", Toast.LENGTH_SHORT).show();
                startRotation();
            }
        });
    }

    private void startRotation() {
        runnable = new Runnable() {
            @Override
            public void run() {
                changeWallpaper();
                // Set the next one in 30 seconds
                handler.postDelayed(this, 30000);
            }
        };
        handler.post(runnable);
    }

    private void changeWallpaper() {
        WallpaperManager myWallpaperManager = WallpaperManager.getInstance(getApplicationContext());
        int randomIndex = random.nextInt(images.length);
        int resId = images[randomIndex];

        try {
            myWallpaperManager.setResource(resId);
            Log.d(TAG, "Wallpaper changed to resource: " + resId);
        } catch (IOException e) {
            Log.e(TAG, "Failed to set wallpaper: " + e.getMessage());
            Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show();
        }
    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
        if (handler != null && runnable != null) {
            handler.removeCallbacks(runnable);
        }
    }
}

MainActivity.java


package com.example.btn_str_chn;

import android.os.Bundle;
import android.os.Handler;
import android.widget.Button;
import android.widget.TextView;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {
    Handler handler=new Handler();
    int i=1;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        Button start1=findViewById(R.id.btn_start);
        TextView tv=findViewById(R.id.textview1);
        Button stop1=findViewById(R.id.btn_stop);
        Runnable r=new Runnable() {
            @Override
            public void run() {
                tv.setText(""+i++);
                handler.postDelayed(this,1000);
            }
        };
        start1.setOnClickListener(v->handler.post(r));
        stop1.setOnClickListener(v->handler.removeCallbacks(r));
    }
}
